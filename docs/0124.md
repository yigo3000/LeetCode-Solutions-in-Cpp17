* 遍历每个节点，计算该节点到叶子节点的最长路径和，在此过程中记录最大的总路径和
```cpp
class Solution {
public:
    int maxPathSum(TreeNode* root) {
        dfs(root);
        return res;
    }

    int dfs(TreeNode* root)
    {
        if (!root) return 0;
        int l = max(dfs(root->left), 0);
        int r = max(dfs(root->right), 0);
        res = max(res, l + r + root->val);
        return root->val + max(l, r);
    }
private:
    int res = INT_MIN;
};
```
