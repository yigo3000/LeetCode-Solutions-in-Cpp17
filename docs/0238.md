* 把当前数左边所有数的乘积和右边两个数的乘积保存到两个数组中，同时遍历两个数组，得到的乘积即为其他元素的乘积
```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n = size(nums);
        vector<int> l(n, 1);
        vector<int> r(n, 1);
        for (int i = 1; i < n; ++i)
        {
            l[i] = l[i - 1] * nums[i - 1];
        }
        for (int i = n - 2; i >= 0; --i)
        {
            r[i] = r[i + 1] * nums[i + 1];
        }
        for (int i = 0; i < n; ++i)
        {
            l[i] *= r[i];
        }
        return l;
    }
};
```
