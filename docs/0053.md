* 遍历每个数，计算以当前数为子数组尾元素的最大和。当上次的和为负数时，则从当前数开始重新计算和
```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int sz = size(nums);
        int res = INT_MIN;
        int sum = 0;
        for (auto& x : nums)
        {
            if (sum >= 0)
            {
                sum += x;
            }
            else
            {
                sum = x;
            }
            res = max(res, sum);
        }
        return res;
    }
};
```
