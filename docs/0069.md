* 二分查找即可，注意判断过程中，值的平方可能产生溢出，因此判断应该用除法而非乘法
```cpp
while (l < r)
{
    int m = l + (r - l) / 2;
    int target = x / m; // m * m可能溢出
    if (m == target)
    {
        return m;
    }
    else if (m > target)
    {
        r = m;
    }
    else
    {
        l = m + 1;
    }
}
return l;
```
* 上面求得的是第一个满足`x / m < m`的值，但商应该是向下取整，因此最后应该返回`l - 1`
```cpp
class Solution {
public:
    int mySqrt(int x) {
        if (x <= 1) return x;
        int l = 1;
        int r = x;
        while (l < r)
        {
            int m = l + (r - l) / 2;
            int target = x / m;
            if (m == target)
            {
                return m;
            }
            else if (m > target)
            {
                r = m;
            }
            else
            {
                l = m + 1;
            }
        }
        return l - 1;
    }
};
```
