* 将节点存储到队列中，每次循环弹出一层，同时把下一层添加到队列中，队列为空则层次遍历结束
```cpp
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        queue<TreeNode*> q;
        q.emplace(root);
        while (!empty(q))
        {
            vector<int> tmp;
            int n = size(q);
            while (n--)
            {
                TreeNode* t = q.front();
                q.pop();
                tmp.emplace_back(t->val);
                if (t->left) q.emplace(t->left);
                if (t->right) q.emplace(t->right);
            }
            res.emplace_back(tmp);
        }
        return res;
    }
};
```
