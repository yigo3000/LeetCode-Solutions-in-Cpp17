* 设置左右边界，从左向右遍历
* 遇到0放到左边界并将左边界右移
* 遇到2放到右边界并将右边界左移
* 遇到1不用操作，继续访问下一个数
* 遍历完成后0全在左侧，2全在右侧，中间就是1
```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int cur = 0;
        int l = 0;
        int r = size(nums);
        while (cur < r)
        {
            if (nums[cur] == 0 && nums[l] == 0)
            {
                ++cur;
                ++l;
            }
            else if (nums[cur] == 0)
            {
                swap(nums[cur], nums[l++]);
            }
            else if (nums[cur] == 2)
            {
                swap(nums[cur], nums[--r]);
            }
            else
            {
                ++cur;
            }
        }
    }
};
```

class Solution {

        {
            if (nums[cur] == 0)
            {
                swap(nums[cur++], nums[l++]);
            }
            else if (nums[cur] == 2)
            {
                swap(nums[cur], nums[--r]);
            }
            else
            {
                ++cur;
            }
        }
    }
};
