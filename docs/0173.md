* 用栈保存中序遍历左侧序列即可
```cpp
class BSTIterator {
public:
    BSTIterator(TreeNode* root) {
        while (root)
        {
            s.emplace(root);
            root = root->left;
        }
    }
    
    /** @return the next smallest number */
    int next() {
        TreeNode* t = s.top();
        s.pop();
        int res = t->val;
        t = t->right;
        while (t)
        {
            s.emplace(t);
            t = t->left;
        }
        return res;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext() {
        return !empty(s);
    }
private:
    stack<TreeNode*> s;
};
```
