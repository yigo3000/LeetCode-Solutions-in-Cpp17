* 层次遍历后，对第偶数行翻转即可
```cpp
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        queue<TreeNode*> q;
        q.emplace(root);
        while (!empty(q))
        {
            vector<int> tmp;
            int n = size(q);
            while (n--)
            {
                TreeNode* t = q.front();
                q.pop();
                tmp.emplace_back(t->val);
                if (t->left) q.emplace(t->left);
                if (t->right) q.emplace(t->right);
            }
            res.emplace_back(tmp);
        }
        for (int i = 1; i < size(res); i += 2)
        {
            reverse(begin(res[i]), end(res[i]));
        }
        return res;
    }
};
```
* 利用双端队列
```cpp
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        deque<TreeNode*> d{root};
        bool flag = true;
        while (!empty(d))
        {
            vector<int> tmp;
            int n = size(d);
            while (n--)
            {
                if (flag)
                {
                    TreeNode* t = d.back();
                    tmp.emplace_back(t->val);
                    d.pop_back();
                    if (t->left) d.emplace_front(t->left);
                    if (t->right) d.emplace_front(t->right);
                }
                else
                {
                    TreeNode* t = d.front();
                    tmp.emplace_back(t->val);
                    d.pop_front();
                    if (t->right) d.emplace_back(t->right);
                    if (t->left) d.emplace_back(t->left);
                }
            }
            flag = !flag;
            res.emplace_back(tmp);
        }
        return res;
    }
};
```
